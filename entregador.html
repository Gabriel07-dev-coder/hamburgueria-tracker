// Variáveis de controle let mostrarGanhos = true; let saldoAcumulado = 0; // Escuta os ganhos no Firebase onSnapshot(doc(db, "entregadores", "entregador_joao"), (docSnap) => { const dados = docSnap.data(); if (dados && dados.ganhos !== undefined) { saldoAcumulado = parseFloat(dados.ganhos); atualizarInterfaceGanhos(); } }); // Atualiza a tela do entregador function atualizarInterfaceGanhos() { const elementoTexto = document.getElementById('txt-ganhos'); const elementoIcone = document.getElementById('eye-icon'); if (elementoTexto && elementoIcone) { if (mostrarGanhos) { elementoTexto.innerText = `R$ ${saldoAcumulado.toLocaleString('pt-BR', {minimumFractionDigits: 2})}`; elementoIcone.innerText = "👁️"; } else { elementoTexto.innerText = "R$ ••••"; elementoIcone.innerText = "🙈"; } } } // Clique no botão de ganhos document.getElementById('btn-ganhos').onclick = () => { mostrarGanhos = !mostrarGanhos; atualizarInterfaceGanhos(); };
