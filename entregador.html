// Variáveis de controle de exibição let mostrarGanhos = true; let saldoAcumulado = 0; // Escuta específica para os ganhos no documento do João Gabriel onSnapshot(doc(db, "entregadores", "entregador_joao"), (docSnap) => { const dados = docSnap.data(); // Verifica se o campo 'ganhos' existe no Firebase if (dados && dados.ganhos !== undefined) { saldoAcumulado = parseFloat(dados.ganhos); atualizarInterfaceGanhos(); } }); // Função para atualizar o texto e o ícone do olho function atualizarInterfaceGanhos() { const elementoTexto = document.getElementById('txt-ganhos'); // Certifique-se que o ID no HTML é este const elementoIcone = document.getElementById('eye-icon'); // Certifique-se que o ID no HTML é este if (mostrarGanhos) { elementoTexto.innerText = `R$ ${saldoAcumulado.toLocaleString('pt-BR', {minimumFractionDigits: 2})}`; elementoIcone.innerText = "👁️"; } else { elementoTexto.innerText = "R$ ••••"; elementoIcone.innerText = "🙈"; } } // Evento de clique na pílula de ganhos para ocultar/exibir document.getElementById('btn-ganhos').onclick = () => { mostrarGanhos = !mostrarGanhos; atualizarInterfaceGanhos(); };
