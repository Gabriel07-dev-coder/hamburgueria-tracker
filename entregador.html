// Variáveis de controle de exibição let mostrarGanhos = true; let saldoAcumulado = 0; // Escuta específica para os ganhos no Firebase onSnapshot(doc(db, "entregadores", "entregador_joao"), (docSnap) => { const dados = docSnap.data(); if (dados && dados.ganhos !== undefined) { saldoAcumulado = parseFloat(dados.ganhos); atualizarInterfaceGanhos(); } }); // Função para atualizar o texto e o ícone function atualizarInterfaceGanhos() { const elementoTexto = document.getElementById('txt-ganhos'); const elementoIcone = document.getElementById('eye-icon'); if (mostrarGanhos) { elementoTexto.innerText = `R$ ${saldoAcumulado.toLocaleString('pt-BR', {minimumFractionDigits: 2})}`; elementoIcone.innerText = "👁️"; } else { elementoTexto.innerText = "R$ ••••"; elementoIcone.innerText = "🙈"; } } // Evento de clique na pílula document.getElementById('btn-ganhos').onclick = () => { mostrarGanhos = !mostrarGanhos; atualizarInterfaceGanhos(); };
