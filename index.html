import { monitorarEntregador } from './app.js'; // 1. Configuração de Ícones (Diferentes para não confundir) const motoIcon = L.icon({ iconUrl: 'https://cdn-icons-png.flaticon.com/512/2972/2972185.png', iconSize: [45, 45], iconAnchor: [22, 45] }); const entregaIcon = L.icon({ iconUrl: 'https://cdn-icons-png.flaticon.com/512/581/581640.png', iconSize: [35, 35], iconAnchor: [17, 35] }); // 2. Inicializa o mapa const map = L.map('map').setView([-25.4431, -49.2761], 14); L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map); // 3. Função para adicionar entrega clicando no mapa map.on('click', function(e) { const { lat, lng } = e.latlng; const nomeCliente = prompt("Nome do Cliente ou Número do Pedido:"); if (nomeCliente) { L.marker([lat, lng], { icon: entregaIcon }) .addTo(map) .bindPopup(`Entrega: ${nomeCliente}`) .openPopup(); console.log(`Nova entrega marcada em: ${lat}, ${lng}`); } }); let marker; // 4. Monitora a ANA (ID fixo que você já está usando) monitorarEntregador('8zlr1Gb8EDj6JdRXHfNR', (data) => { const { lat, lng, nome } = data; if (!marker) { marker = L.marker([lat, lng], { icon: motoIcon }).addTo(map).bindPopup(nome).openPopup(); } else { marker.setLatLng([lat, lng]); } });
